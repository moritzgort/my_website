listings_clean <- listings %>% 
  filter(name != "",
         host_location != "") %>% 
  mutate(price = parse_number(price),
         bathrooms = parse_number(gsub( "\\s.*", "", bathrooms_text))) %>% 
  filter(bathrooms != "")

listings_clean <- listings_clean %>% 
  filter(bedrooms <= 2,
         beds <= 2,
         minimum_nights <= 4,
         maximum_nights >= 4,
         price <= 1500, # Everything above that would be extraordinarily expensive
